Code Examples

/* Basic CSS Syntax */
h1 {
    color: #264de4;
    font-size: 2.5rem;
    text-align: center;
}

.button {
    background-color: #ffd700;
    padding: 10px 20px;
    border-radius: 5px;
    border: none;
}
/* Flexbox Example */
.container {
    display: flex;
    justify-content: center;
    align-items: center;
    gap: 20px;
    flex-wrap: wrap;
}
/* Responsive Design */
@media (max-width: 768px) {
    .container {
        flex-direction: column;
    }
    
    h1 {
        font-size: 1.5rem;
    }
}
